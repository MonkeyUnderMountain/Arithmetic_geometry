\section{Arithmetic Hilbert-Samuel theorem}


    \begin{theorem}\label{thm:arithmetic_Hilbert_Samuel}
        Let \(\calX\) be an arithmetic variety of absolute dimension \(d\) over \(\Spec \calO_K\), 
        and let \(\overline{\calL}\) be a hermitian line bundle on \(\calX\). 
        Assume that \(\overline{\calL}\) is nef. 
        Then, we have
        \[
            \hvol(\overline{\calL}) = \adeg\left( \widehat{c}_1(\overline{\calL})^d \right).
        \]
    \end{theorem}
